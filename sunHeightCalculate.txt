import ephem

# Konumu belirtin (örneğin İstanbul)
observer = ephem.Observer()
observer.lat = '41.0082'
observer.lon = '28.9784'

# Tarih ve zamanı belirtin (örneğin 2023-08-10 12:00:00)
date_time = '2023-08-10 12:00:00'

# Güneş nesnesini oluşturun
sun = ephem.Sun()

# Tarih ve zamana göre güneşin yüksekliğini hesaplayın
observer.date = date_time
sun.compute(observer)

# Güneşin yüksekliğini alın (radyan cinsinden)
sun_altitude = sun.alt

# Radyanı dereceye çevirin
sun_altitude_deg = ephem.degrees(sun_altitude).get_real()

print(f"Güneş Yüksekliği: {sun_altitude_deg} derece")
